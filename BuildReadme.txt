#To Build the UI from the command line
#It's important to note that you must build from the ProjectAutograph Directory for packages to work properly
javac ./autograph/ui/mainWindow.java
#Then simply run with
java ./autograph/ui/mainWindow

#To Build the Graph Library from the command line
#It's important to note that you must build from the ProjectAutograph Directory for packages to work properly
javac ./autograph/Autograph.java
#Then simply run with 
java ./autograph/Autograph
